209 void captureMoves(board_t playerBoard, board_t opponentBoard,
210  board_t emptyBoard, board_t piece, char type, char color,
211  board_t *movesList, int depth, DIRECTION lockedDirection) {
212  board_t move;
213  board_t tmp;
214  for (size_t d = NW; d <= SE; d++) {
215  if (d != lockedDirection) {
216  if (type == 'k') {
217  tmp = piece;
218  board_t start = piece;
219  move = piece;
220  while ((tmp = shift(move, d)) & emptyBoard)
221  move = tmp;
222  if (shift((shift(move, d) & opponentBoard), d) & emptyBoard) {
223  tower_t *oldTower, *newTower;
224  oldTower = getTower(start);
225  newTower = getTower(move);
226  moveTower(&oldTower, &newTower);
227  playerBoard = (playerBoard & ~start) | move;
228  emptyBoard = ~(playerBoard | opponentBoard);
229  piece = move;
230  }
231  }
232  move = shift((shift(piece, d) & opponentBoard), d) & emptyBoard;
233  if (move) {
234  tower_t *oldTower, *newTower;
235  tmp = piece;
236  oldTower = getTower(tmp);
237  newTower = getTower(move);
238  board_t captured = shift(move, turnAroundDirection(d));
239  char capturedTowerColor =
240  capture(&oldTower, &newTower, getTower(captured));
241  playerBoard = (playerBoard & ~tmp) | move;
242  opponentBoard = opponentBoard & ~captured;
243  if (capturedTowerColor == color) {
244  playerBoard |= captured;
245  } else if (capturedTowerColor != 0) {
246  opponentBoard |= captured;
247  }
248  emptyBoard = ~(playerBoard | opponentBoard);
249  movesList[depth] |= move;
250  if ((color == 'b' && (move & WHITE_BASELINE)) ||
251  (color == 'w' && (move & BLACK_BASELINE)))
252  type = 'k';
253  captureMoves(playerBoard, opponentBoard, emptyBoard, move, type, color,
254  movesList, depth + 1, turnAroundDirection(d));
255  break;
256  }
257  }
258  }

261 board_t *calculateMoves(board_t playerBoard, board_t opponentBoard,
262  board_t piece, char type, char color,
263  bool *captureMovesOnly) {
264  static board_t *movesList;
265  if (movesList == NULL) {
266  movesList = calloc(12, sizeof(board_t));
267  } else {
268  for (size_t i = 0; i < 12; i++) {
269  movesList[i] = 0ULL;
270  }
271  }
272  board_t emptyBoard = ~(playerBoard | opponentBoard);
273  if (type == 'n') {
274  captureMoves(playerBoard, opponentBoard, emptyBoard, piece, type, color,
275  movesList, 0, UNDEFINED);
276  if (movesList[0] == 0ULL && !(*captureMovesOnly)) {
277  if (color == 'w') {
278  movesList[0] = (shift(piece, NW) | shift(piece, NE)) & emptyBoard;
279  } else if (color == 'b') {
280  movesList[0] = (shift(piece, SW) | shift(piece, SE)) & emptyBoard;
281  }
282  } else {
283  *captureMovesOnly = true;
284  }
285  } else if (type == 'k') {
286  captureMoves(playerBoard, opponentBoard, emptyBoard, piece, type, color,
287  movesList, 0, UNDEFINED);
288  if (movesList[0] == 0ULL && !(*captureMovesOnly)) {
289  board_t move;
290  board_t nonCaptureMoves = 0ULL;
291  board_t tmp;
292  for (size_t d = NW; d <= SE; d++) {
293  move = piece;
294  while ((tmp = shift(move, d)) & emptyBoard) {
295  move = tmp;
296  nonCaptureMoves |= move;
297  }
298  }
299  movesList[0] = nonCaptureMoves;
300  } else {
301  *captureMovesOnly = true;
302  }
303  }
304  return movesList;
